Problem 4

Write a JavaScript program to find the area of a triangle where three sides are 5, 6, 7.

Description: This program calculates the area of a triangle using Heron's formula, given the lengths of its three sides.


        function triangleArea(a, b, c) {
            const s = (a + b + c) / 2; // Calculate the semi-perimeter
            const area = Math.sqrt(s * (s - a) * (s - b) * (s - c)); // Calculate the area using Heron's formula
            return area;
        }
        
        const side1 = 5;
        const side2 = 6;
        const side3 = 7;
        
        const area = triangleArea(side1, side2, side3);
        console.log("The area of the triangle is:", area);

Problem 5

Write a JavaScript program to rotate the string 'w3resource' in the right direction. This is done by periodically removing one letter from the string end and attaching it to the front.

Description: This program rotates the string 'w3resource' to the right by removing the last character and appending it to the front, repeatedly at set intervals.


        function rotateString(str) {
            return str[str.length - 1] + str.slice(0, str.length - 1); // Rotate string by moving last character to the front
        }
        
        let rotatingString = 'w3resource';
        
        setInterval(() => {
            rotatingString = rotateString(rotatingString);
            console.log(rotatingString); // Log the rotated string every second
        }, 1000);

Problem 6

Write a JavaScript program to determine whether a given year is a leap year in the Gregorian calendar.

Description: This program checks if a given year is a leap year by following the rules of the Gregorian calendar.


        function isLeapYear(year) {
            // A year is a leap year if it is divisible by 4, but not by 100, unless it is also divisible by 400
            if (year % 4 === 0) {
                if (year % 100 === 0) {
                    if (year % 400 === 0) {
                        return true;
                    } else {
                        return false;
                    }
                } else {
                    return true;
                }
            } else {
                return false;
            }
        }
        
        const year = 2024; // Example year
        console.log("Is " + year + " a leap year?", isLeapYear(year)); // Check if the year is a leap year

Problem 7

Write a JavaScript program to find out if 1st January will be a Sunday between 2014 and 2050.

Description: This program checks each year between 2014 and 2050 to determine if January 1st falls on a Sunday.


        function findSundays() {
            for (let year = 2014; year <= 2050; year++) {
                let date = new Date(year, 0, 1); // Create a date object for January 1st of the current year
                if (date.getDay() === 0) { // Check if the day is Sunday (0 represents Sunday)
                    console.log("1st January is a Sunday in the year " + year);
                }
            }
        }
        
        findSundays(); // Call the function to find and log years when January 1st is a Sunday
